    {
        $signature = $_GET["signature"];

        $nonce = $_GET["nonce"];

        $tmpArr = array($token, $timestamp, $nonce);

        $tmpStr = implode( $tmpArr );
        $tmpStr = sha1( $tmpStr );
    
        if ($tmpStr == $signature ) {
            exit($_GET['echostr']);

    }
